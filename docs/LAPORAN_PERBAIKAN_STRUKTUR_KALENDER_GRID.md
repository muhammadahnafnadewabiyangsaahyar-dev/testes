function generateMonthView(month, year) {
    const calendarBody = document.getElementById('calendar-body');
    
    if (!calendarBody) return;
    
    calendarBody.innerHTML = '';
    
    const firstDay = new Date(year, month, 1).getDay();
    const daysInMonth = new Date(year, month + 1, 0).getDate();
    
    // Get unique employees from shift assignments
    const currentEmployees = Object.values(shiftAssignments || {}).map(a => ({
        id: a.user_id,
        name: a.nama_lengkap || a.pegawai_name || 'Unknown'
    }));
    
    const uniqueEmployees = [...new Map(currentEmployees.map(emp => [emp.id, emp])).values()]
        .sort((a, b) => a.name.localeCompare(b.name));

    // Generate rows for each employee
    uniqueEmployees.forEach(employee => {
        const row = document.createElement('tr');
        
        // Employee name cell (first column)
        const employeeCell = document.createElement('td');
        employeeCell.textContent = employee.name;
        employeeCell.classList.add('employee-name-cell');
        row.appendChild(employeeCell);
        
        // Generate 7 days of the week
        let rowDate = 1;
        for (let week = 0; week < 6; week++) {
            for (let day = 0; day < 7; day++) {
                const cell = document.createElement('td');
                cell.classList.add('calendar-day-cell');
                
                // Calculate if this cell should be empty
                const isBeforeFirstDay = week === 0 && day < firstDay;
                const isAfterLastDay = rowDate > daysInMonth;
                const isOtherMonth = isBeforeFirstDay || isAfterLastDay;
                
                if (isOtherMonth) {
                    cell.classList.add('other-month');
                } else {
                    // Create day number
                    const dateDiv = document.createElement('div');
                    dateDiv.className = 'calendar-day-number';
                    dateDiv.textContent = rowDate;
                    cell.appendChild(dateDiv);
                    
                    // Add click handler to switch to day view
                    const currentDateForCell = new Date(year, month, rowDate);
                    cell.addEventListener('click', function() {
                        currentDate = currentDateForCell;
                        switchView('day');
                    });
                    
                    // Show shift assignments for this employee and date
                    const dateStr = window.KalenderUtils.formatDate(new Date(year, month, rowDate));
                    const shiftsForEmployeeDate = Object.values(shiftAssignments).filter(
                        assignment => assignment.user_id === employee.id && 
                                    assignment.tanggal_shift === dateStr
                    );
                    
                    if (shiftsForEmployeeDate.length > 0) {
                        const shiftsDiv = document.createElement('div');
                        shiftsDiv.className = 'day-events';
                        
                        shiftsForEmployeeDate.forEach(assignment => {
                            const shiftEvent = document.createElement('div');
                            shiftEvent.className = 'shift-event';
                            
                            // Determine shift type and style
                            let shiftClass = 'shift-pagi';
                            let shiftLabel = assignment.nama_shift || assignment.shift_type || 'Shift';
                            
                            if (shiftLabel.toLowerCase().includes('sore') || shiftLabel.toLowerCase().includes('middle')) {
                                shiftClass = 'shift-sore';
                            } else if (shiftLabel.toLowerCase().includes('malam')) {
                                shiftClass = 'shift-malam';
                            }
                            
                            shiftEvent.classList.add(shiftClass);
                            shiftEvent.textContent = shiftLabel;
                            
                            shiftsDiv.appendChild(shiftEvent);
                        });
                        
                        cell.appendChild(shiftsDiv);
                    }
                    
                    rowDate++;
                }
                
                row.appendChild(cell);
            }
            if (rowDate > daysInMonth) break;
        }
        
        calendarBody.appendChild(row);
    });
}
```

function generateWeekView(date) {
    // Calculate week start (Monday)
    const weekStart = new Date(date);
    const day = weekStart.getDay();
    const diff = weekStart.getDate() - day + (day === 0 ? -6 : 1);
    weekStart.setDate(diff);
    
    // Clear and rebuild week view
    const weekCalendar = document.getElementById('week-calendar');
    if (weekCalendar) {
        weekCalendar.innerHTML = '';
        
        // Create week grid
        const weekGrid = document.createElement('div');
        weekGrid.className = 'week-calendar';
        
        // Create time column
        const timeColumn = document.createElement('div');
        timeColumn.className = 'week-time-column';
        
        // Add time slots (24 hours)
        for (let hour = 0; hour < 24; hour++) {
            const timeSlot = document.createElement('div');
            timeSlot.className = 'week-time-slot';
            timeSlot.textContent = `${String(hour).padStart(2, '0')}:00`;
            timeColumn.appendChild(timeSlot);
        }
        
        // Create days column
        const daysColumn = document.createElement('div');
        daysColumn.className = 'week-days-column';
        
        for (let i = 0; i < 7; i++) {
            const currentDay = new Date(weekStart);
            currentDay.setDate(weekStart.getDate() + i);
            
            // Day header
            const dayHeader = document.createElement('div');
            dayHeader.className = 'week-day-header';
            const dayNames = ['Minggu', 'Senin', 'Selasa', 'Rabu', 'Kamis', 'Jumat', 'Sabtu'];
            dayHeader.textContent = `${dayNames[currentDay.getDay()]} ${currentDay.getDate()}`;
            
            // Day cells
            for (let hour = 0; hour < 24; hour++) {
                const dayCell = document.createElement('div');
                dayCell.className = 'week-day-cell';
                
                // Add click handler to open day view for specific time
                dayCell.addEventListener('click', function() {
                    const selectedDate = new Date(currentDay);
                    selectedDate.setHours(hour);
                    currentDate = selectedDate;
                    switchView('day');
                });
                
                dayHeader.appendChild(dayCell);
            }
            
            daysColumn.appendChild(dayHeader);
        }
        
        weekGrid.appendChild(timeColumn);
        weekGrid.appendChild(daysColumn);
        weekCalendar.appendChild(weekGrid);
    }
}
```

```javascript
function generateYearView(year) {
    const yearGrid = document.getElementById('year-grid');
    if (!yearGrid) return;
    
    yearGrid.innerHTML = '';
    
    const monthNames = window.KalenderUtils.monthNames;
    
    for (let month = 0; month < 12; month++) {
        const monthDiv = document.createElement('div');
        monthDiv.className = 'year-month';
        
        const monthTitle = document.createElement('h4');
        monthTitle.textContent = monthNames[month];
        monthDiv.appendChild(monthTitle);
        
        const miniTable = document.createElement('table');
        const thead = document.createElement('thead');
        const tbody = document.createElement('tbody');
        
        // Table header
        const headerRow = document.createElement('tr');
        const dayHeaders = ['M', 'S', 'S', 'R', 'K', 'J', 'S'];
        dayHeaders.forEach(day => {
            const th = document.createElement('th');
            th.textContent = day;
            th.style.cssText = 'font-size: 11px; color: #666; text-align: center; padding: 4px;';
            headerRow.appendChild(th);
        });
        thead.appendChild(headerRow);
        
        // Mini calendar for each month
        const firstDay = new Date(year, month, 1).getDay();
        const adjustedFirstDay = firstDay === 0 ? 6 : firstDay - 1;
        const daysInMonth = new Date(year, month + 1, 0).getDate();
        
        // Add empty cells before first day
        for (let i = 0; i < adjustedFirstDay; i++) {
            const emptyCell = document.createElement('td');
            emptyCell.style.cssText = 'padding: 2px; background: #f5f5f5;';
            tbody.appendChild(emptyCell);
        }
        
        // Add day cells
        for (let date = 1; date <= daysInMonth; date++) {
            const dayCell = document.createElement('td');
            dayCell.textContent = date;
            dayCell.style.cssText = 'text-align: center; padding: 4px; cursor: pointer; border-radius: 3px; transition: background-color 0.2s;';
            
            // Highlight today
            const today = new Date();
            const isToday = date === today.getDate() && 
                           month === today.getMonth() && 
                           year === today.getFullYear();
            
            if (isToday) {
                dayCell.classList.add('today');
            }
            
            dayCell.addEventListener('click', () => {
                currentDate = new Date(year, month, date);
                currentMonth = month;
                currentYear = year;
                switchView('month');
            });
            
            dayCell.addEventListener('mouseenter', function() {
                if (!isToday) {
                    this.style.backgroundColor = '#e3f2fd';
                }
            });
            
            dayCell.addEventListener('mouseleave', function() {
                if (!isToday) {
                    this.style.backgroundColor = '';
                }
            });
            
            // Check if this is a weekend (Saturday and Sunday)
            const cellDate = new Date(year, month, date);
            const dayOfWeek = cellDate.getDay();
            if (dayOfWeek === 0 || dayOfWeek === 6) {
                dayCell.style.color = '#999';
            }
            
            tbody.appendChild(dayCell);
        }
        
        miniTable.appendChild(thead);
        miniTable.appendChild(tbody);
        monthDiv.appendChild(miniTable);
        yearGrid.appendChild(monthDiv);
    }
}
```
